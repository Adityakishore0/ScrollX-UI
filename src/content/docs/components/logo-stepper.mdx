## Usage

```tsx
import { LogoStepper } from "@/components/ui/logo-stepper";
```

```tsx
<LogoStepper
  logos={logos}
  direction="loop"
  animationDelay={1.2}
  animationDuration={0.6}
  visibleCount={5}
/>
```
