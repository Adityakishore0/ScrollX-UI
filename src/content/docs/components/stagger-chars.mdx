## Usage

### Basic Usage

```tsx
import { StaggerChars } from "@/components/ui/stagger-chars";

export function MyComponent() {
  return (
    <StaggerChars
      text="HOVER ME"
      className="text-white"
      fromClassName="text-white"
      toClassName="text-blue-400"
    />
  );
}
```

## Easing Presets

The component includes several easing presets for different animation feels:

```tsx
import { EASING_PRESETS } from "@/components/ui/stagger-chars";

// Available presets:
EASING_PRESETS.smooth; // [0.22, 1, 0.36, 1]
EASING_PRESETS.bouncy; // [0.68, -0.55, 0.265, 1.55]
EASING_PRESETS.sharp; // [0.4, 0, 0.2, 1]
EASING_PRESETS.gentle; // [0.25, 0.46, 0.45, 0.94]
EASING_PRESETS.linear; // [0, 0, 1, 1]
EASING_PRESETS.easeIn; // [0.42, 0, 1, 1]
EASING_PRESETS.easeOut; // [0, 0, 0.58, 1]
EASING_PRESETS.easeInOut; // [0.42, 0, 0.58, 1]
```
