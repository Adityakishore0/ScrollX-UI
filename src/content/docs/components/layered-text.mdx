## Usage

```tsx
import { LayeredText } from "@/components/ui/layered-text";
```

```tsx
<LayeredText
  text="ScrollX UI"
  className="text-4xl md:text-6xl font-bold text-black dark:text-white [--stroke-color:#FFFFFF] dark:[--stroke-color:#000000]"
  layers={["#06B6D4", "#14B8A6", "#84CC16"]}
  offsetX={2}
  offsetY={2}
  strokeWidth={2}
  animate={true}
/>
```
