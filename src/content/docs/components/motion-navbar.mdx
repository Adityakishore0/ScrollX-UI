## Usage

```tsx
import {
  Navbar,
  NavBody,
  NavItems,
  NavDropdown,
  MobileNav,
  NavbarLogo,
  NavbarButton,
  MobileNavHeader,
  MobileNavToggle,
  MobileNavMenu,
} from "@/components/ui/motion-navbar";
```

```tsx
<Navbar className="bg-black dark:bg-white text-white dark:text-black">
      <NavbarLogo />
      <NavBody>
        <NavItems items={navItems} />
        <NavDropdown label="resources" items={dropdownItems} />
        <NavbarButton href="/get-started" variant="primary">
          get started
        </NavbarButton>
      </NavBody>

      <MobileNav>
        <MobileNavHeader>
          <MobileNavToggle isOpen={false} onClick={() => {}} />
        </MobileNavHeader>
        <MobileNavMenu>
          <NavItems items={navItems} />
          <NavDropdown label="resources" items={dropdownItems} />
          <NavbarButton href="/get-started" variant="primary">
            get started
          </NavbarButton>
        </MobileNavMenu>
      </MobileNav>
    </Navbar>

```
