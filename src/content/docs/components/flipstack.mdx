## Usage

```tsx
import FlipStack from "@/components/ui/flipstack";
```

```tsx
<FlipStack
  cards={[
    { id: 1, content: "Card One" },
    { id: 2, content: "Card Two" },
    { id: 3, content: "Card Three" },
  ]}
/>
```
