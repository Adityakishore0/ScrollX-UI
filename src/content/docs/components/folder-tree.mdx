## Usage

```tsx
import { FolderTree } from "@/components/ui/folder-tree";
```

### Basic Example

```tsx
const folderData = {
  id: 'root',
  name: 'Root',
  type: 'folder',
  children: [
    {
      id: 'folder1',
      name: 'Documents',
      type: 'folder',
      children: [
        {
          id: 'file1',
          name: 'Document 1',
          type: 'file',
          extension: 'docx'
        }
      ]
    },
    {
      id: 'file2',
      name: 'README',
      type: 'file',
      extension: 'md'
    }
  ]
};

<FolderTree 
  data={folderData} 
  initialExpanded={['root']}
/>
```
